🗂️ Tables to SELECT FROM

📝 otel.otel_logs
Primary table for all Claude Code events and telemetry
Use for: Sessions, tools, errors, performance, user activity
🔗 otel.otel_traces
Distributed tracing data for request flows
Use for: Request tracing, span analysis, performance debugging
📊 otel.otel_metrics_*
Metrics tables (gauge, histogram, sum, summary)
Use for: Quantitative measurements, counters, histograms
💾 otel.logs
Alternative logs view with different schema
Use for: Alternative log analysis, backup data source
📈 otel.daily_trace_summary
Aggregated daily trace summaries
Use for: Daily reports, trend analysis, performance summaries

🔍 WHERE Clause Modifiers

⏰ Time Filters
WHERE Timestamp >= now() - INTERVAL 1 HOUR
WHERE Timestamp >= now() - INTERVAL 1 DAY
WHERE Timestamp >= now() - INTERVAL 1 WEEK
WHERE TimestampTime BETWEEN '2025-09-01' AND '2025-09-03'
WHERE toDate(Timestamp) = today()
👤 User & Session Filters
WHERE LogAttributes['user.email'] = 'user@example.com'
WHERE LogAttributes['session.id'] = 'abc123'
WHERE LogAttributes['user.id'] IS NOT NULL
WHERE ServiceName = 'claude-code'
🛠️ Tool & Event Filters
WHERE LogAttributes['tool_name'] = 'Bash'
WHERE LogAttributes['success'] = 'true'
WHERE Body = 'claude_code.tool_result'
WHERE LogAttributes['event.name'] LIKE '%error%'
📊 Performance Filters
WHERE LogAttributes['duration_ms'] > 5000
WHERE LogAttributes['cost_usd'] > 0.01
WHERE LogAttributes['cache_hit'] = 'true'
WHERE LogAttributes['token_count'] > 1000
🚨 Error Filters
WHERE Body = 'claude_code.api_error'
WHERE SeverityText = 'ERROR'
WHERE LogAttributes['error_type'] IS NOT NULL
WHERE LogAttributes['success'] = 'false'
💬 Message & Content Filters
WHERE Body = 'claude_code.user_prompt'
WHERE Body = 'claude_code.api_request'
WHERE LogAttributes['prompt'] IS NOT NULL
WHERE LogAttributes['prompt_length'] > 500
WHERE LogAttributes['input_tokens'] > 1000
WHERE LogAttributes['output_tokens'] > 100
WHERE LogAttributes['cache_read_tokens'] > 0
WHERE LogAttributes['model'] LIKE '%sonnet%'
🔧 Advanced Patterns
WHERE LogAttributes['tool_name'] IN ('Bash', 'Read', 'Write')
WHERE LogAttributes['duration_ms'] BETWEEN 1000 AND 10000
WHERE hasAll(LogAttributes, ['session.id', 'tool_name'])
WHERE length(LogAttributes['error_message']) > 100

💡 Complete Query Examples

📊 Session Analysis
SELECT LogAttributes['user.email'] as user, COUNT(DISTINCT LogAttributes['session.id']) as sessions, SUM(toFloat64OrNull(LogAttributes['cost_usd'])) as total_cost FROM otel.otel_logs WHERE Timestamp >= now() - INTERVAL 7 DAY GROUP BY LogAttributes['user.email'] ORDER BY total_cost DESC
🛠️ Tool Performance
SELECT LogAttributes['tool_name'] as tool, COUNT(*) as usage_count, AVG(toInt64OrNull(LogAttributes['duration_ms'])) as avg_duration_ms, COUNT(CASE WHEN LogAttributes['success'] = 'true' THEN 1 END) as success_count FROM otel.otel_logs WHERE LogAttributes['tool_name'] IS NOT NULL AND Timestamp >= now() - INTERVAL 1 DAY GROUP BY LogAttributes['tool_name'] ORDER BY usage_count DESC
🚨 Error Analysis
SELECT LogAttributes['error_type'] as error_type, COUNT(*) as error_count, MIN(Timestamp) as first_occurrence, MAX(Timestamp) as last_occurrence, COUNT(DISTINCT LogAttributes['session.id']) as affected_sessions FROM otel.otel_logs WHERE Body = 'claude_code.api_error' AND Timestamp >= now() - INTERVAL 24 HOUR GROUP BY LogAttributes['error_type'] ORDER BY error_count DESC
💬 Message Content Analysis
SELECT LogAttributes['user.email'] as user, Body as message_type, COUNT(*) as message_count, AVG(toInt64OrNull(LogAttributes['prompt_length'])) as avg_prompt_length, SUM(toInt64OrNull(LogAttributes['input_tokens'])) as total_input_tokens, SUM(toInt64OrNull(LogAttributes['output_tokens'])) as total_output_tokens, SUM(toFloat64OrNull(LogAttributes['cost_usd'])) as total_cost, LogAttributes['model'] as model_used FROM otel.otel_logs WHERE Body IN ('claude_code.user_prompt', 'claude_code.api_request') AND Timestamp >= now() - INTERVAL 7 DAY GROUP BY LogAttributes['user.email'], Body, LogAttributes['model'] ORDER BY message_count DESC
